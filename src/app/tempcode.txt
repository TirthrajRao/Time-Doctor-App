  dataURItoBlob(dataURI: string): Observable<Blob> {
    return Observable.create((observer: Observer<Blob>) => {
      const byteString: string = atob(dataURI);
      const arrayBuffer: ArrayBuffer = new ArrayBuffer(byteString.length);
      const int8Array: Uint8Array = new Uint8Array(arrayBuffer);
      for (let i = 0; i < byteString.length; i++) {
        int8Array[i] = byteString.charCodeAt(i);
      }
      const blob = new Blob([int8Array], { type: "image/png" });
      observer.next(blob);
      observer.complete();
    });
  }

  b64toBlob(b64Data, contentType?, sliceSize?) {
    contentType = contentType || '';
    sliceSize = sliceSize || 512;

    const byteCharacters = atob(b64Data);
    const byteArrays = [];

    for (let offset = 0; offset < byteCharacters.length; offset += sliceSize) {
      const slice = byteCharacters.slice(offset, offset + sliceSize);

      const byteNumbers = new Array(slice.length);
      for (let i = 0; i < slice.length; i++) {
        byteNumbers[i] = slice.charCodeAt(i);
      }

      const byteArray = new Uint8Array(byteNumbers);

      byteArrays.push(byteArray);
    }

    const blob = new Blob(byteArrays, { type: contentType });
    return blob;
  }

  calculateDifference(currentAttendanceLog) {
    console.group('calculateDifference', currentAttendanceLog);

    var in1 = currentAttendanceLog.timeLog[currentAttendanceLog.timeLog.length - 1].in;
    var out = currentAttendanceLog.timeLog[currentAttendanceLog.timeLog.length - 1].out;
    var inn = moment(in1, 'hh:mm:ss: a').diff(moment().startOf('day'), 'seconds');
    var outt = moment(out, 'hh:mm:ss: a').diff(moment().startOf('day'), 'seconds');

    let seconds = outt - inn;
    if (currentAttendanceLog.difference != "-") {
      var difference = moment(currentAttendanceLog.difference, 'hh:mm:ss: a').diff(moment().startOf('day'), 'seconds');

      seconds = seconds + difference;
      console.log("seconds++++++",seconds)
    }

			console.log("PARMAS IN UPDAETE FREE RESOURCE", params, params.freeGrp.members);
			let index = _.findIndex(params.freeGrp.members,  (o) => { return params.userDetails._id != null ?  o.toString() == params.userDetails._id.toString() : false });

			if (index != -1) {
				let ind = _.findIndex(params.removedTeam,  (o)  => { return params.userDetails._id != null ?  o.toString() == params.userDetails._id.toString() : false });
				if (ind != -1) {
					console.log("Found in removed team @ index ::", ind);
					resolve();
				}
				else {
					console.log("Found in add team @ index ::", ind);

					// Remove member from free resource group.

					let objToSend = {
						removedTeam: [params.userDetails._id],
						groupId: params.freeGrp._id
					}

					groupDao.updateGroup(objToSend)
						.then((groupUpdated) => {
							resolve();
							// console.log(groupUpdated);
						})
						.catch((groupUpdateErr) => {
							console.log("groupUpdateErr ==>", groupUpdateErr);
							reject(groupUpdateErr)
						});

				}
			}
			else {

				// If user's group length is 0 and user's id is in removed group arrray. Then add that user in free resource group.
				// console.	log(params.userDetails.group.length == 0, params.removedTeam.includes(params.userDetails._id.toString()));
				if (params.userDetails.group.length == 0 && params.removedTeam.includes(params.userDetails._id.toString())) {

					let objToSend = {
						members: [params.userDetails._id],
						groupId: params.freeGrp._id
					}

					groupDao.updateGroup(objToSend)
						.then((groupUpdated) => {
							resolve();
							// console.log(groupUpdated);
						})
						.catch((groupUpdateErr) => {
							console.log("groupUpdateErr ==>", groupUpdateErr);
							reject(groupUpdateErr);
						});

				}
				else {
					console.log("in else if false false");
					resolve();
				}
			}
		}
		catch (err) {
			console.log(err);
		}
	});
